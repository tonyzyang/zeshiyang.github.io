    df=get_close_price(code_list[0],st,en)
    for i in range(1,len(code_list)):
        dfi=get_close_price(code_list[i],st,en)
        df = pd.merge(df,dfi, on='')
    return df

from numpy import cumsum, log, polyfit, sqrt, std, subtract
from numpy.random import randn
def hurst(ts):
    lags = range(2, 100)
    tau = [sqrt(std(subtract(ts[lag:], ts[:-lag]))) for lag in lags]
    # Use a linear fit to estimate the Hurst Exponent
    poly = polyfit(log(lags), log(tau), 1)
    # Return the Hurst exponent from the polyfit output
    return poly[0]*2.0

def plot_price_series(df, ts1, ts2):
    months = mdates.MonthLocator() 
    fig, ax = plt.subplots()
    ax.plot(df.index, df[ts1], label=ts1)
    ax.plot(df.index, df[ts2], label=ts2)
    ax.xaxis.set_major_locator(months)
    ax.xaxis.set_major_formatter(mdates.DateFormatter('%b %Y'))
    #ax.set_xlim(datetime.datetime(2012, 1, 1), datetime.datetime(2013, 1, 1))
    ax.grid(True)
    fig.autofmt_xdate()
    plt.xlabel('Month/Year')
    plt.ylabel('Price ($)')
    plt.title('%s and %s Daily Prices' % (ts1, ts2))
    plt.legend()
    plt.show()

    plt.xlabel('%s Price ($)' % ts1)
    plt.ylabel('%s Price ($)' % ts2)
    plt.title('%s and %s Price Scatterplot' % (ts1, ts2))

    plt.show()

    ax.grid(True)
    fig.autofmt_xdate()

    plt.title('Residual Plot')

    plt.show()

def create_lagged_series(symbol, st, en, lags=5):
    ts = tu.pro_api().daily(ts_code=symbol, start_date=st, end_date=en).drop(['ts_code'], axis=1).sort_values(by='trade_date')
     # Create the new lagged DataFrame
    tslag = pd.DataFrame(index=ts['trade_date'])
    tslag["Today"] = ts["pct_chg"].tolist()
    tslag["Volume"] = ts["vol"].tolist()
    # Create the shifted lag series of prior trading period close values
    for i in range(0, lags):
        tslag["Lag%s" % str(i+1)] = ts["pct_chg"].shift(1+i).tolist()
        tsret = pd.DataFrame(index=tslag.index)
        tsret["Volume"] = tslag["Volume"]
        tsret["Today"] = tslag["Today"]
    for i,x in enumerate(tsret["Today"]):
        if (abs(x) < 0.0001):
             tsret["Today"][i] = 0.0001
            # Create the lagged percentage returns columns
    for i in range(0, lags):
        tsret["Lag%s" % str(i+1)] = \
        tslag["Lag%s" % str(i+1)]
        # Create the "Direction" column (+1 or -1) indicating an up/down day
    tsret["Direction"] = np.sign(tsret["Today"])
    tsret = tsret[tsret.index >= st]
    tsret=tsret.dropna(axis=0,how='any')
    return tsret

def annualised_sharpe(returns, N=252):
    return np.sqrt(N) * returns.mean() / returns.std()
def stock_sharpe_ratio(symbol,st,en):
    pdf = tu.pro_api().daily(ts_code=symbol, start_date=st, end_date=en).drop(['ts_code'], axis=1).sort_values(by='trade_date')
    # Use the percentage change method to easily calculate daily returns
    pdf['daily_ret'] = pdf['pct_chg']
    pdf['excess_daily_ret'] = pdf['daily_ret'] - 0.05/252
    # Return the annualised Sharpe ratio based on the excess daily returns
    return annualised_sharpe(pdf['excess_daily_ret'])

from scipy.stats import norm
def var_cov_var(P, c, mu, sigma):
    alpha = norm.ppf(1-c, mu, sigma)
    return P - P*(alpha + 1)
