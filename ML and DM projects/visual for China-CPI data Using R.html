cpi<-cpi[,-1]
cpii<-ts(cpi, frequency = 12, start = c(2014, 1))
#cpii is return data, so make it to 0.XX
cpii<-round(cpii/100,3)
#draw a line first
plot(cpii[,1],main="CPI change%", ylab="increase %")
#add the other 2 lines
lines(cpii[,2],lwd=2,col="red")
lines(cpii[,3],lwd=2,lty=4, col="blue")
#set the chart legend
legend(x="bottomright",legend=c(colnames(cpii)[1],colnames(cpii)[2],colnames(cpii)[3]),
       col=c("black","red","blue"),lty=c(1,1,4))

#plot in one time,seperate in 3 charts automatically
